Requete selection table besoin :
1 - WIN

select bsn.title, bsn.dateCreation,clt.name, bsn.statut
from besoin as bsn
INNER JOIN client as clt
on clt.id = bsn.id_client
WHERE bsn.statut = 'win'

2 - LOST

select bsn.title, bsn.dateCreation,clt.name, bsn.statut
from besoin as bsn
INNER JOIN client as clt
on clt.id = bsn.id_client
WHERE bsn.statut = 'lost'

3 - OPEN

select bsn.title, bsn.dateCreation,clt.name, bsn.statut
from besoin as bsn
INNER JOIN client as clt
on clt.id = bsn.id_client
